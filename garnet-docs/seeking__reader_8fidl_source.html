9 // Reader with seek semantics.
10 interface SeekingReader {
11  const uint64 kUnknownSize = 0xffffffffffffffff;
12 
13  // Gets the content size and whether the reader can seek. The |result|
14  // callback parameter is |OK| unless there was a problem accessing the
15  // content to be read. Other possible values:
16  // |NOT_FOUND| - The content could not be found.
17  // |UNKNOWN_ERROR| - Some other error occurred.
18  // A |size| value of |kUnknownSize| if the size of the content isn't known.
19  Describe() => (MediaResult result, uint64 size, bool can_seek);
20 
21  // Reads into a socket starting at a specified position. If the reader
22  // can't seek, position must be 0. The |result| callback parameter is |OK|
23  // unless there was a problem performing the read. Other possible values:
24  // |NOT_FOUND| - The content could not be found.
25  // |INVALID_ARGUMENT| - The reader can't seek and |position| was non-zero.
26  // |UNKNOWN_ERROR| - Some other error occurred.
27  ReadAt(uint64 position) => (MediaResult result, handle<socket>? socket);
28 };
