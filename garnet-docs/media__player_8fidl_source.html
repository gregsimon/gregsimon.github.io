12 // Plays media.
13 interface MediaPlayer {
14  // Special value for GetStatus version_last_seen parameter to get the current
15  // status immediately.
16  const uint64 kInitialStatus = 0;
17 
18  // Starts playback.
19  Play();
20 
21  // Pauses playback.
22  Pause();
23 
24  // Seeks to the specified position, specified in nanoseconds.
25  Seek(int64 position);
26 
27  // Gets the status. To get the status immediately, call
28  // GetStatus(kInitialStatus). To get updates thereafter, pass the version
29  // sent in the previous callback.
30  GetStatus(uint64 version_last_seen)
31  => (uint64 version, MediaPlayerStatus status);
32 
33  // Sets the reader.
34  SetReader(SeekingReader? reader);
35 };

37 // MediaPlayer status information.
38 struct MediaPlayerStatus {
39  // Transform translating local time to presentation time. Reverse translation
40  // (presentation time to local time) is only valid when media is playing.
41  TimelineTransform? timeline_transform;
42 
43  // Indicates whether presentation has reached end-of-stream.
44  bool end_of_stream;
45 
46  // Whether the current content has an audio stream.
47  bool content_has_audio;
48 
49  // Whether the current content has a video stream.
50  bool content_has_video;
51 
52  // Indicates whether an audio stream is currently connected for rendering.
53  // This value will be false if |content_has_audio| is false, if no audio
54  // renderer was provided, or if the audio media type isn't supported.
55  bool audio_connected;
56 
57  // Indicates whether a video stream is currently connected for rendering.
58  // This value will be false if |content_has_video| is false, if no video
59  // renderer was provided, or if the video media type isn't supported.
60  bool video_connected;
61 
62  // Describes the media.
63  MediaMetadata? metadata;
64 
65  // Indicates a problem preventing intended operation.
66  Problem? problem;
67 };
