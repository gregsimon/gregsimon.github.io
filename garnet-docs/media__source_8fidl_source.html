12 // Produces media streams delivered from a specified origin.
13 interface MediaSource {
14  // Special value for GetStatus version_last_seen parameter to get the current
15  // status immediately.
16  const uint64 kInitialStatus = 0;
17 
18  // Describes the source.
19  Describe() => (array<MediaType> stream_types);
20 
21  // Gets the packet producer for the specified stream.
22  GetPacketProducer(uint32 stream_index, MediaPacketProducer& packet_producer);
23 
24  // Gets the status. To get the status immediately, call
25  // GetStatus(kInitialStatus). To get updates thereafter, pass the version
26  // sent in the previous callback.
27  GetStatus(uint64 version_last_seen)
28  => (uint64 version, MediaSourceStatus status);
29 
30  // Flushes the source and downstream components. |hold_frame| indicates
31  // whether a video renderer (if any) should continue to display the last
32  // displayed frame.
33  Flush(bool hold_frame) => ();
34 
35  // Seeks to the specified position, specified in nanoseconds.
36  Seek(int64 position) => ();
37 };

39 // Describes the status of the media source.
40 struct MediaSourceStatus {
41  // Indicates whether an audio stream producer is currently connected. This
42  // value will be false if the content contains no audio or if no audio stream
43  // producer is connected (possibly because the audio stream media type isn't
44  // supported).
45  bool audio_connected;
46 
47  // Indicates whether a video stream producer is currently connected. This
48  // value will be false if the content contains no video or if no video stream
49  // producer is connected (possibly because the video stream media type isn't
50  // supported).
51  bool video_connected;
52 
53  // Describes the media.
54  MediaMetadata? metadata;
55 
56  // Indicates a problem preventing intended operation.
57  Problem? problem;
58 };
