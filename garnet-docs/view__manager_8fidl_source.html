11 // The view manager manages trees of views which represent user interface
12 // components.
13 //
14 // Applications create views to describe user interface components
15 // to present content to the user and support user interaction.
16 //
17 // The system creates a view tree to hold the root of a view hierarchy
18 // containing views from various applications.
19 //
20 // Refer to |View| and |ViewTree| for more information about these objects.
21 [ServiceName="mozart::ViewManager"]
22 interface ViewManager {
23  // Maximum length for a view or view tree label.
24  const uint32 kLabelMaxLength = 32;
25 
26  // Gets the scene manager associated with this view manager.
27  // All graphical content for this view manager's views and view trees
28  // must come from sessions created by this |scene_manager|.
29  GetSceneManager(scenic.SceneManager& scene_manager);
30 
31  // Creates a view.
32  //
33  // The |view| is used to configure the view and interact with its
34  // local environment.
35  //
36  // The |view_owner| is used as a transferable reference which can
37  // be passed to the view's intended container as part of a request to
38  // add the view as a child. The view manager itself does not describe
39  // how this interaction should take place, only that ownership should
40  // eventually be passed back through the container's view interface
41  // as an argument to |View.AddChild()|.
42  //
43  // The |view_listener| is used to receive events from the view.
44  //
45  // The |label| is an optional name to associate with the view for
46  // diagnostic purposes. The label will be truncated if it is longer
47  // than |kLabelMaxLength|.
48  //
49  // |parent_export_token| is an export token which the view manager
50  // will use to export the node to which the view's content should be
51  // attached.
52  //
53  // To present graphical content, the view must obtain a |Session| from
54  // the |SceneManager|, create an event pair, bind one endpoint to an
55  // |ImportResourceOp| using |ImportSpec.NODE|, attach its content nodes as
56  // descendants of the imported node, and pass the other endpoint of the
57  // event pair to this method as |parent_export_token|.
58  //
59  // To destroy the view and cause it to be removed from the view tree,
60  // simply close the |view|, |view_listener|, or |view_owner| channels.
61  CreateView(View& view,
62  ViewOwner& view_owner,
63  ViewListener view_listener,
64  handle<eventpair> parent_export_token,
65  string? label);
66 
67  // Creates a view tree.
68  //
69  // The |view_tree| is used to configure the view tree and interact
70  // with the views it contains.
71  //
72  // The |view_tree_listener| is used to receive events from the view tree.
73  //
74  // The |label| is an optional name to associate with the view tree for
75  // diagnostic purposes. The label will be truncated if it is longer
76  // than |kLabelMaxLength|.
77  //
78  // To destroy the view tree simply close the |view_tree| or
79  // |view_tree_listener| channels.
80  CreateViewTree(ViewTree& view_tree,
81  ViewTreeListener view_tree_listener,
82  string? label);
83 };
