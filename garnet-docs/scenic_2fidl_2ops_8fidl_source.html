74 // Releases the client's reference to the resource; it is then illegal to use
75 // the ID in subsequent Ops. Other references to the resource may exist, so
76 // releasing the resource does not result in its immediate destruction; it is
77 // only destroyed once the last reference is released. For example, the
78 // resource may be required to render an in-progress frame, or it may be
79 // referred to by another resource). However, the ID will be immediately
80 // unregistered, and may be reused to create a new resource.
81 struct ReleaseResourceOp {
82  // ID of the resource to be dereferenced.
83  uint32 id;
84 };

86 // Create an external reference to the specified resource, which can then be
87 // imported into another Session by passing a handle to |token|'s peer to
88 // ImportResourceOp; see that comment for more details.
89 //
90 // The importing client is typically in a different process than the exporter.
91 // No specific mechanism is provided for transferring a token from an exporter
92 // to an importer; collaborators may choose any out-of-band API they wish to do
93 // so.
94 struct ExportResourceOp {
95  uint32 id;
96  handle<eventpair> token;
97 };
98 
99 // Import a resource that was exported via ExportResourceOp(). |token| is a
100 // handle to the eventpair peer that was used to export the resource, and |spec|
101 // describes the type of the imported resource, and the ops which can legally be
102 // applied to it. Afterward, |id| can be used to refer to the resource in an
103 // Op, similarly (but not identically: see below) to a resource that was created
104 // in the session. For example, you can add children to an imported EntityNode
105 // via AddChildOp.
106 //
107 // However, note that the importer does not gain full access to the imported
108 // resource, but rather to an attenuated subset of its capabilities. For
109 // example, you cannot use a DetachOp to detach an imported EntityNode from its
110 // parent.
111 //
112 // Unlike ExportResourceOp, there is no configurable timeout. There is an
113 // expectation that the exported resource will become available in a short
114 // amount of time. TODO: this needs elaboration... e.g. we might notify via the
115 // SessionListener when we know that the link will never be made (e.g. if the
116 // peer of the import token is destroyed).
117 //
118 // TODO: describe how the imported resource behaves if the exported resource
119 // isn't yet available, or becomes unavailable (e.g. an imported Material might
120 // act as a plain white texture).
121 struct ImportResourceOp {
122  uint32 id;
123  handle<eventpair> token;
124  ImportSpec spec;
125 };

159 // Add a node as a part of another node. The implications of being a part
160 // rather than a child differ based on the type of the part. However, one
161 // implication is constant: removing all of a node's children (e.g. via
162 // DetachChildrenOp) does not affect its parts. This is similar to the "shadow

164 // using the shadow DOM, and do no show up amongst the children of that element.

172 // was applied first.

301 // Sets/clears a node's clip.
302 //
303 // Constraints:
304 // - |node_id| refs a |Node| with the has_clip characteristic.
305 // - |clip_id| a |Node| with the is_clip characteristic, or nothing. If the
306 // referenced node is not rooted, then it will have no effect (since its
307 // full world-transform cannot be determined).
308 // - |clip_to_self| If false, children are only clipped to the region specified
309 // by |clip_id|. If true, children are additionally clipped to the node's
310 // shape (as determined by its ShapeNode parts).
311 //
312 // Discussion:
313 // If a node has a clip, it will be applied to both the parts and the children
314 // of the node. Under some circumstances (TBD), a clip will not be applicable
315 // to a node; in such cases it will be as though no clip has been specified for
316 // the node.
317 // TODO: elaborate on the circumstances under which a clip is inapplicable.
318 // For example, consider a 3D space that looks through a portal into a 2D space
319 // that uses a clip for a circular reveal. It would not be meaningful to clip
320 // objects on the outside (i.e. in the 3D space).
321 struct SetClipOp {
322  uint32 node_id;
323  uint32 clip_id;
324  bool clip_to_self;
325 };

405 // Set a mesh's indices and vertices.
406 //
407 // |mesh_id| refs the Mesh to be updated.
408 // |index_buffer_id| refs a Buffer that contains the mesh indices.
409 // |index_format| defines how the index buffer data is to be interpreted.
410 // |index_offset| number of bytes from the start of the index Buffer.
411 // |index_count| number of indices.
412 // |vertex_buffer_id| refs a Buffer that contains the mesh vertices.
413 // |vertex_format| defines how the vertex buffer data is to be interpreted.
414 // |vertex_offset| number of bytes from the start of the vertex Buffer.
415 // |vertex_count| number of vertices.
416 // |bounding_box| must contain all vertices within the specified range.
417 //
418 // The MeshVertexFormat defines which per-vertex attributes are provided by the
419 // mesh, and the size of each attribute (and therefore the size of each vertex).
420 // The attributes are ordered within the vertex in the same order that they
421 // appear within the MeshVertexFormat struct. For example, if the values are
422 // kVector3, kNone and kVector2, then:
423 // - each vertex has a position and UV-coordinates, but no surface normal.
424 // - the 3D position occupies bytes 0-11 (3 dimensions * 4 bytes per float).
425 // - the UV coords occupy bytes 12-19, since no surface normal is provided.
426 enum MeshIndexFormat {
427  // TODO(MZ-275): only kUint32 is currently supported.
428  kUint16 = 1,
429  kUint32,
430 };
431 struct MeshVertexFormat {
432  // kVector2 or kVector3.
433  ValueType position_type;
434  // kVector2 or kVector3 (must match position_type), or kNone.
435  ValueType normal_type;
436  // kVector2 or kNone.
437  ValueType tex_coord_type;
438 };
