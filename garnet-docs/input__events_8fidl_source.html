10 // |KeyboardEvent| represents event generated by a user's interaction with a
11 // keyboard.
12 //
13 // Those events are triggered by distinct pressed state changes of the keys.
14 //
15 // The state transitions should be as follows:
16 // PRESSED -> (REPEAT ->) RELEASED
17 // or
18 // PRESSED -> (REPEAT ->) CANCELLED
19 //
20 // The input system will repeat those events automatically when a code_point is
21 // available.
22 struct KeyboardEvent {
23  // Time the event was delivered. The time is in nanoseconds and corresponds
24  // to the uptime of the machine.
25  uint64 event_time;
26 
27  uint32 device_id;
28 
29  enum Phase {
30  // When key is pressed down.
31  PRESSED,
32  // When key is released.
33  RELEASED,
34  // This key |PRESSED| is not directed to this input client anymore.
35  CANCELLED,
36  // Whether this is an automatically generated key repeat
37  REPEAT,
38  };
39  Phase phase;
40 
41  // Keyboard HID Usage
42  // See http://www.usb.org/developers/hidpage/Hut1_12v2.pdf
43  uint32 hid_usage;
44 
45  // The unicode code point represented by this key event, if any.
46  // Dead keys are represented as Unicode combining characters.
47  //
48  // If there is no unicode code point, this value is zero.
49  uint32 code_point;
50 
51  // Key modifiers as defined by the different kModifier constants such as
52  // |kModifierCapsLock| currently pressed
53  uint32 modifiers;
54 };

56 // Pointers represent raw data about the user's interaction with the screen.
57 //
58 // The state transitions should be as follows:
59 // ADD (-> HOVER) -> DOWN -> MOVE -> UP (-> HOVER) -> REMOVE
60 //
61 // At any point after the initial ADD, a transition to CANCEL is also possible.
62 struct PointerEvent {
63  // Time the event was delivered. The time is in nanoseconds and corresponds
64  // to the uptime of the machine.
65  uint64 event_time;
66 
67  uint32 device_id;
68 
69  uint32 pointer_id;
70 
71  enum Type {
72  // A touch-based pointer device.
73  TOUCH,
74 
75  // A pointer device with a stylus.
76  STYLUS,
77 
78  // A pointer device with a stylus that has been inverted.
79  INVERTED_STYLUS,
80 
81  // A pointer device with a stylus.
82  MOUSE,
83  };
84  Type type;

86  enum Phase {
87  // The device has started tracking the pointer.
88  //
89  // For example, the pointer might be hovering above the device, having not yet
90  // made contact with the surface of the device.
91  ADD,
92 
93  // The pointer has moved with respect to the device while not in contact with
94  // the device.
95  HOVER,
96 
97  // The pointer has made contact with the device.
98  //
99  // For |MOUSE| devices, this is triggered when the primary button is pressed
100  // down to emulate a touch on the screen.
101  DOWN,
102 
103  // The pointer has moved with respect to the device while in contact with the
104  // device.
105  MOVE,
106 
107  // The pointer has stopped making contact with the device.
108  //
109  // For |MOUSE| devices, this is triggered when the primary button is
110  // released.
111  UP,
112 
113  // The device is no longer tracking the pointer.
114  //
115  // For example, the pointer might have drifted out of the device's hover
116  // detection range or might have been disconnected from the system entirely.
117  REMOVE,
118 
119  // The input from the pointer is no longer directed towards this receiver.
120  CANCEL,
121 
122  // TODO: add phases to indicate button press / release
123  };
124  Phase phase;
