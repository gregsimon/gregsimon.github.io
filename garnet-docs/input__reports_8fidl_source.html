7 // Descriptors are used to describe the capabilities of an input device.
8 //
9 // Devices can have multiple descriptors of the following kinds
10 // |KeyboardDescriptor|, |MouseDescriptor|, |StylusDescriptor| and
11 // |TouchscreenDescriptor|
12 //
13 // An input device will generate |InputReport| corresponding to the
14 // capabilities of that device listed in its descriptor.
15 // For instance a input device with a |KeyboardDescriptor| will generate
16 // |KeyboardReport| when a key is pressed on the keyboard.
17 
18 // Describe a |Range| of values
19 struct Range {
20  int32 min;
21  int32 max;
22 };
23 
24 struct RangeF {
25  float min;
26  float max;
27 };

64 // |MouseDescriptor| describes the capabilities of a mouse.
65 struct MouseDescriptor {
66  // The range of relative X and Y movement which can be described by a mouse
67  // report.
68  Axis rel_x;
69  Axis rel_y;
70 
71  // The range of relative vertical and horizontal scroll which can be
72  // described by a mouse report.
73  Axis? vscroll;
74  Axis? hscroll;
75 
76  // The list of HID mouse usages that this mouse can generate.
77  uint32 buttons;
78 };
79 
80 // |MouseReport| gives the relative mouvement of the mouse and currently
81 // pressed buttons.
82 struct MouseReport {
83  // Relative X and Y positional displacement.
84  int32 rel_x;
85  int32 rel_y;
86 
87  // Relative horizontal and vertical scrolling displacement.
88  int32 rel_hscroll;
89  int32 rel_vscroll;
90 
91  // buttons currently down

97 // |Stylus| describes the capabilities of a stylus.
98 struct StylusDescriptor {
99  // Ranges for the |x| and |y| axis of the stylus.
100  Axis x;
101  Axis y;
102 
103  // Range for the pressure of the tip
104  Axis? pressure;
105 
106  bool is_invertible = false;
107 
108  // The list of HID button usages that this stylus can generate.
109  uint32 buttons;
110 };
111 
112 // |StylusReport| describes the current state of the stylus.
113 struct StylusReport {
114  // Current position of the stylus within the range described in
115  // |StylusDescriptor|
116  int32 x;
117  int32 y;
118 
119  // Pressure applied on the stylus tip
120  uint32 pressure;
121 
122  // Whether the stylus has made contact with the surface.
123  bool is_in_contact;
124 
125  // Whether the stylus is within range. If |is_in_contact| is false, then the stylus
126  // is hovering.
127  bool in_range;
128 
129  // Whether the stylus is thought to be inverted.
130  bool is_inverted;
131 
132  // List of buttons currently pressed down.
133  uint32 pressed_buttons;
134 };

146 // |Touch| describes one touch on a touchscreen, which should correspond to
147 // one finger.
148 struct Touch {
149  // Identifier for a finger that is down.
150  // Note: |finger_id| might not be sequential and will range from 0 to
151  // |max_finger_id|
152  uint32 finger_id;
153 
154  // Location within the axis defined in |TouchscreenDescriptor|
155  int32 x;
156  int32 y;
157 
158  // Area pressed.
159  uint32 width;
160  uint32 height;
161 };
162 
163 // |TouchscreenReport| describes the current touches recorded by the touchscreen
164 // and holds a |Touch| per finger down.
165 struct TouchscreenReport {
