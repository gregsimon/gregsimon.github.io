39 // A reader that reads messages from one or more logs.
40 interface FlogReader {
41  // Gets entries from the log starting and the specified index (entries are
42  // indexed starting at 0). If the log is open, the callback will be called
43  // when max_count entries are avaiable starting at start_index. If the log
44  // is closed, the callback will be called immediately with as many entries
45  // as are available starting at start_index and not exceeding max_count
46  // entries.
47  GetEntries(uint32 start_index, uint32 max_count) =>
48  (array<FlogEntry> entries);
49 };

58 // Log entry produced by |FlogReader|. Entry type is determined by interrogating
59 // the |details| field. FlogEntry and its subordinates duplicate the semantics
60 // of FlogLogger. This approach keeps producers simple (they don't have to
61 // create a bunch of structs and unions to log a message) and allows consumers
62 // to pull entries rather than having to implement FlogLogger.
63 struct FlogEntry {
64  int64 time_ns;
65  uint32 log_id;
66  uint32 channel_id;
67  FlogEntryDetails? details;
68 };
