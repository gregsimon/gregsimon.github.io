7 // Information returned by methods such as |ImagePipe.PresentImage()| and
8 // |Session.Present()|, when the consumer begins preparing the first frame
9 // which includes the presented content.
10 struct PresentationInfo {
11  // The actual time at which the enqueued operations are anticipated to take
12  // visible effect, expressed in nanoseconds in the |CLOCK_MONOTONIC|
13  // timebase.
14  //
15  // This value increases monotonically with each new frame, typically in
16  // increments of the |presentation_interval|.
17  uint64 presentation_time;
18 
19  // The nominal amount of time which is anticipated to elapse between
20  // successively presented frames, expressed in nanoseconds. When rendering
21  // to a display, the interval will typically be derived from the display
22  // refresh rate.
23  //
24  // This value is non-zero. It may vary from time to time, such as when
25  // changing display modes.

27 };
