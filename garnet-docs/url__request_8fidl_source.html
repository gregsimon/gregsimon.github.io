10 struct URLRequest {
11  // Specify the cache behavior of the request.
12  enum CacheMode {
13  // Default behavior.
14  DEFAULT,
15 
16  // The HTTP request will bypass the local cache and will have a
17  // 'Cache-Control: nocache' header added in that causes any proxy servers
18  // to also not satisfy the request from their cache. This has the effect
19  // of forcing a full end-to-end fetch.
20  BYPASS_CACHE,
21 
22  // The HTTP request will fail if it cannot serve the requested resource
23  // from the cache (or some equivalent local store).
24  ONLY_FROM_CACHE,
25  };
26 
27  // Specify the mechanism used to return the response body.
28  enum ResponseBodyMode {
29  // The complete response body should be returned in the |buffer| field of
30  // the response body.
31  BUFFER,
32  // The response body should be streamed through the |stream| field of the
33  // response body.
34  STREAM,
35  // The response body may be returned as a buffer or stream.
36  BUFFER_OR_STREAM,
37  };
38 
39  // The URL to load.
40  string url;
41 
42  // The HTTP method if applicable.
43  string method = "GET";
44 
45  // Additional HTTP request headers.
46  array<HttpHeader>? headers;
47 
48  // The payload for the request body. For HTTP requests, the method must be set
49  // to "POST" or "PUT".
50  URLBody? body;
51 
52  // The buffer size of the socket returned in URLResponse's |body| member.
53  // A value of 0 indicates that the default buffer size should be used. This
54  // value is just a suggestion. The URLLoader may choose to ignore this value.
55  uint32 response_body_buffer_size = 0;
56 
57  // If set to true, then redirects will be automatically followed. Otherwise,
58  // when a redirect is encounterd, FollowRedirect must be called to proceed.
59  bool auto_follow_redirects = false;
60 
61  // The cache behavior for the request.
62  CacheMode cache_mode = DEFAULT;
63 
64  // The response body mode.
65  ResponseBodyMode response_body_mode = STREAM;
66 };
