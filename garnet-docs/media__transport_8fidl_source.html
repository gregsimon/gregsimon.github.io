10 // Describes a fragment of media content, including metadata and a locator for
11 // the payload.
12 struct MediaPacket {
13  const int64 kNoTimestamp = 0x7fffffffffffffff;
14 
15  // Presentation timestamp. Time at which the media should be presented,
16  // according to the media timeline.
17  int64 pts = kNoTimestamp;
18 
19  // Indicates the number of PTS ticks that correspond to pts_rate_seconds
20  // seconds. In other words, the time represented by a single PTS tick in
21  // seconds is pts_rate_ticks / pts_rate_seconds. For example, if PTS is
22  // given in nanoseconds, pts_rate_ticks would be 1,000,000,000, and
23  // pts_rate_seconds would be 1.
24  uint32 pts_rate_ticks;
25 
26  // Indicates the number of seconds that correspond to pts_rate_ticks
27  // PTS ticks. See pts_rate_ticks for more info.
28  uint32 pts_rate_seconds;
29 
30  // Indicates whether this packet is a keyframe.
31  bool keyframe;
32 
33  // Indicates whether this is the last packet in the stream.
34  bool end_of_stream;
35 
36  // Identifier of the shared buffer that contains the payload for this packet.
37  // This identifier must be established by a call to MediaPacketConsumer's
38  // AddPayloadBuffer method before it can be used.
39  uint32 payload_buffer_id;
40 
41  // Offset of the packet payload in the indicated shared buffer.
42  uint64 payload_offset;
43 
44  // Size in bytes of the packet payload.
45  uint64 payload_size;
46 
47  // Revised media type describing this and subsequent packets.
48  MediaType? revised_media_type;
49 };

51 // Describes consumer demand for MediaPackets. Demand is initially assumed to
52 // be the default values given below. Changes in demand can be signalled
53 // in-band with SupplyPacket responses or by responding to PullDemandUpdate.
54 struct MediaPacketDemand {
55  // The minimum number of packets that should be outstanding at any moment.
56  // The producer should try to keep at least this many SupplyPacket operations
57  // pending.
58  uint32 min_packets_outstanding = 0;
59 
60  // The minimum packet PTS value. If the last packet supplied has a PTS less
61  // than this value, another packet should be supplied. A value of kNoTimestamp
62  // means this value is not applicable.
63  int64 min_pts = MediaPacket.kNoTimestamp;
64 };

66 // Models a stream producer. A MediaPacketProducer allows a client to connect
67 // the producer to a MediaPacketConsumer so packets flow from the producer to
68 // the consumer.
69 //
70 // The client calls Connect to connect producer and consumer. The producer then
71 // calls PushPacket on the consumer to deliver packets.
72 interface MediaPacketProducer {
73  // Connects this MediaPacketProducer to a MediaPacketConsumer.
74  Connect(MediaPacketConsumer consumer) => ();
75 
76  // Disconnects this MediaPacketProducer from a previously-connected
77  // MediaPacketConsumer.
78  Disconnect();
79 };

81 // Models a stream consumer. A MediaPacketConsumer allows a client to send
82 // packets directly to the consumer or to connect the consumer to a
83 // MediaPacketProducer so packets flow from the producer to the consumer.
84 //
85 // In the former scenario, the client calls PushPacket to deliver a packet. The
86 // callback notifies the client that the consumer is done with the packet
87 // buffer region.
88 //
89 // In the latter scenario, the client calls Connect on the producer to connect
90 // producer and consumer. The producer then calls PushPacket on the consumer to
91 // deliver packets.
92 interface MediaPacketConsumer {
93  const uint64 kMaxBufferLen = 0x3FFFFFFFFFFFFFFF;
94 
95  // Gets the demand. The consumer is free to respond to this method when it
96  // sees fit to do so. Demand can be communicated in the response to
97  // SupplyPacket, so this method is only used when the demand update can't wait
98  // until the next SupplyPacket response is sent.
99  PullDemandUpdate() => (MediaPacketDemand? demand);
100 
101  // Adds a payload buffer that may subsequently be referenced by a MediaPacket.
102  AddPayloadBuffer(uint32 payload_buffer_id,
103  handle<vmo> payload_buffer);
104 
105  // Indicates that the identified payload buffer will no longer be used.
106  RemovePayloadBuffer(uint32 payload_buffer_id);
107 
108  // Supplies a packet to the consumer. The callback signals that the consumer
109  // is done with the packet and its payload. Demand is only provided in the
110  // response if it has changed.
111  SupplyPacket(MediaPacket packet) => (MediaPacketDemand? demand);
112 
113  // Flushes the stream. The callback signals that the flush operation is
114  // complete. |hold_frame| indicates whether a video renderer (if any) should
115  // continue to display the last displayed frame.
116  Flush(bool hold_frame) => ();
117 };
