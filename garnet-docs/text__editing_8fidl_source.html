7 // Whether a TextPosition is visually upstream or downstream of its offset.
8 //
9 // For example, when a text position exists at a line break, a single offset has
10 // two visual positions, one prior to the line break (at the end of the first
11 // line) and one after the line break (at the start of the second line). A text
12 // affinity disambiguates between those cases. (Something similar happens with
13 // between runs of bidirectional text.)
14 enum TextAffinity {
15  // The position has affinity for the upstream side of the text position.
16  //
17  // For example, if the offset of the text position is a line break, the
18  // position represents the end of the first line.
19  UPSTREAM,
20 
21  // The position has affinity for the downstream side of the text position.
22  //
23  // For example, if the offset of the text position is a line break, the
24  // position represents the start of the second line.
25  DOWNSTREAM,
26 };

41 // A range of text that represents a selection.
42 //
43 // Text selection is always directional. Direction should be determined by
44 // comparing base and extent.
45 struct TextSelection {
46  // The offset at which the selection originates.
47  //
48  // Might be larger than, smaller than, or equal to extent.
49  int64 base;
50 
51  // The offset at which the selection terminates.
52  //
53  // When the user uses the arrow keys to adjust the selection, this is the
54  // value that changes. Similarly, if the current theme paints a caret on one
55  // side of the selection, this is the location at which to paint the caret.
56  //
57  // Might be larger than, smaller than, or equal to base.
58  int64 extent;
59 
60  // If the the text range is collapsed and has more than one visual location
61  // (e.g., occurs at a line break), which of the two locations to use when
62  // painting the caret.
63  TextAffinity affinity;
64 };
