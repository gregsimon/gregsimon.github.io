10 // Stores attributes related to an account that is exposed to device shell.
11 // A list of existing account(s) can be obtained via
12 // UserProvider.PreviousUsers() and a new account can be added via
13 // UserProvider.AddAccount().
14 struct Account {
15  // A randomly generated identifier that is used to identify this
16  // account on this device. This is meant to be used by device shell when it
17  // wants to login as a user who has previously logged in.
18  string id;
19 
20  // The identity provider that was used to authenticate the user on this
21  // device.
22  IdentityProvider identity_provider;
23 
24  // The name that is displayed on the device shell while logging in. Display
25  // name is fetched from user profile attributes as configured by the user at
26  // the given identity provider.
27  string display_name;
28 
29  // User's profile url that is used by the device shell while logging in.
30  // Profile url is fetched from user profile attributes as configured by the
31  // user at the given identity provider.
32  string url;
33 
34  // User's profile image url that is used by the device shell while logging in.
35  // Profile image url is fetched from user profile attributes as configured by
36  // the user at the given identity provider.
37  string image_url;
38 };

53 // An interface that allows the Framework to talk to the token manager service
54 // to add new accounts or be able to get a |TokenProviderFactory| that can
55 // be used to provide specialized instances of |TokenProvider| to third party
56 // agents.
57 [ServiceName="auth::AccountProvider"]
58 interface AccountProvider {
59  // Called by the Framework once, right after starting up a token manager
60  // application.
61  Initialize(AccountProviderContext account_provider_context);
62 
63  // Adds a new user account. This involves connecting to the identity provider
64  // and registering user. Once user has been successfully added, user profile
65  // information such as email, image_url, profile_url, first and last names are
66  // pre-populated in |account| before returning along with Ok status.
67  //
68  // In case of an error, AuthErr status is returned with detailed error code
69  // and message.
70  AddAccount(IdentityProvider identity_provider)
71  => (Account? account, AuthErr status);
72 
73  // Removes an existing user account. This involves talking to account's
74  // identity provider and revoking user credentials both locally and remotely.
75  // This operation also deletes cached tokens for the given account.
76  // TODO(ukode): Modify this api to take account_id and IDP as input once the
77  // Account struct is cleaned up.
78  //
79  // In case of an error, AuthErr status is returned with detailed error code
80  // and message.
81  RemoveAccount(Account account) => (AuthErr status);
82 
83  // Get a |TokenProviderFactory| associated with an |account_id|. It is used
84  // by the Framework to provide a specialized instances of |TokenProvider| to
85  // third party agents.
86  GetTokenProviderFactory(string account_id,
87  TokenProviderFactory& request);
88 };

103 interface AuthenticationContext {
