7 /*
8  * This file contains definitions of data structures used to represent
9  * Fidl types and values.
10  *
11  * As described in the Fidl Language Specification, Fidl types are defined
12  * recursively and consequently a |Type| object may recursively contain
13  * other |Type| objects. For example a |Type| object representing an
14  * array<int32> will contain a |Type| object representing an int32.
15  *
16  * A Fidl type declaration may contain an identifier that resolves to a
17  * user-defined type: a struct, union, enum or interface. We use |TypeReference|
18  * to represent an occurrence of such an identifier. A |TypeReference| may be
19  * resolved or not. Resolved means that the user-defined type to which the
20  * identifier refers has been found and associated with the |TypeReference|.
21  * A |Type| object is fully-resolved if it, and recursively all of its
22  * sub-components, do not contain any unresolved TypeReferences.
23  *
24  * A resolved |TypeReference| does not literally contain a structure
25  * representing the user-defined type that it represents but rather refers to
26  * its target type indirectly via a string called a |type_key|. The type_key may
27  * be used to lookup the user-defined type to which it refers.
28  *
29  * The mapping from |type_keys| to user-defined types is not
30  * represented by any structures in this file and instead must be maintained
31  * by a higher layer context in which this file is used. For example the
32  * |ServiceDescription| interface defined in service_describer.fidl includes
33  * the method:
34  * GetTypeDefinition(string type_key) => UserDefinedType? type);
35  * for this purpose.
36  * We refer to this higher-layer context as the *owning context.*
37  *
38  * In addition to types, Fidl values are also representd by structures in this
39  * file. A |Value| may be a LiteralValue, a ConstantReference,
40  * an EnumValueReference or a BuiltinConstantValue. Similarly to the situation
41  * with TypeReferences, ConstantReferences and EnumValueReferences contain a
42  * key which may be used to lookup user-defined value (an EnumValue or a
43  * DeclaredConstant) in the owning context.
44  */

116 // Represents an occurence of a user-defined type identifier that should
117 // resolve to an enum, struct, union, interface or interface request. This type
118 // reference may be either resolved or unresolved. If the reference has been
119 // resolved then the |type_key| field is populated.
120 struct TypeReference {
121  // Was this occurrence marked as nullable?
122  bool nullable;
123 
124  // Was this occurrence marked as being an interface request? If so then
125  // this reference must (eventually) resolve to an interface type.

127

129  // a short name, a fully-qualified identifier, or a partially qualified
130  // identifier. Either this field or type_key must be non-null. Some
131  // implementations will keep this field even after the reference has been
132  // resolved.
133  // Note(rudominer) Consider requiring this to be the fully-qualified

135  // before a TypeReference structure is created. Otherwise consider adding
136  // a Scope field to TypeReference so that resolution may occur later.
137  string? identifier;
138 
139  // This field is non-null if this reference has been resolved.
140  string? type_key;

376 // This represents a Fidl constant declaration.
377 struct DeclaredConstant {
378  DeclarationData decl_data;
379 
380  // The type must be a StringType, BOOL, a numeric type or a TypeReference
381  // whose resolved type is a FidlEnum.
382  Type type;
383 
384  // This is the value specified in the right-hand-side of the constant
385  // declaration. The value must be one of the following:
386  // (a) a LiteralValue or a BuiltinConstantValue of the same type as |type|
387  // (b) an EnumValueReference that resolves to an EnumValue of the same
388  // enum type as |type|, or
389  // (c) a ConstantReference that resolves to a different
390  // DeclaredConstant whose |resolved_concrete_value| is one of (a) or (b)
391  Value value;
392 
393 
394  // The resolved concrete value. This must be a LiteralValue, a
395  // BuiltinConstantValue, or an EnumValueReference. It may not be a
396  // ConstantReference.
397  //
398  // The resolved concrete value is defined as follows:
399  // If |value| is a LiteralValue, a BuiltinConstantValue or an
400  // EnunValueReference then the resolved
401  // concrete value is conceptually equal to |value| and this will
402  // be indicated by setting the |resolved_concrete_value| field to null.
403  //
404  // Otherwise |value| is a ConstantReference that refers to a different
405  // DeclaredConstant and in this case |resolved_concrete_value| will be
406  // non-null. It will be set to the conceptual resolved concrete value of that
407  // other DeclaredConstant (even if that other declared constant has its own
408  // |resolved_concrete_value| field set to null.) This chain of references
409  // must terminate in well-formed Fidl.
410  //
411  // In the case that |resolved_concrete_value| is set to an EnumValueReference
412  // only the |enum_type_key| and |enum_value_index| fields of the
413  // EnumValueReference are meaningful. The other fields should be ignored.
414  Value? resolved_concrete_value;
415 };

509 // This structure contains the information necessary for an implementation of
510 // |ServiceDescription| (see service_describer.fidl), but restricted to a
511 // single .fidl file and not restricted to a single service.
512 //
513 // Note that this structure is not literally a component of a FidlFileGraph.
514 // Instead, at compile time an instance of this structure is created and
515 // serialized to an array of bytes that is stored in the
516 // |serialized_runtime_type_info| field of each |FidlFile|. Then at code
517 // generation time this array of bytes is written into the generated source
518 // code as a literal array of bytes. Then at runtime this array of bytes
519 // is deserialized into an instance of RuntimeTypeInfo that may be used by
520 // an implementation of |ServiceDescription|.
521 struct RuntimeTypeInfo {
522  // Maps a service name to the type key of the associated interface for all of
523  // the services contained in a single .fidl file. A "service" is an
524  // interface that has been annoted with the "ServiceName" annotation.
525  // This indicates that it can be returned from
526  // ServiceProvider.ProvideService() and its description can be returned
527  // from ServiceDescriber.DescribeService().
528  map<string, string> services;
529 
530  // All of the resolved user-defined-types contained in a single .fidl File.
531  // The keys are the |type_key|s. Note that the complete set of types
532  // referenced (recursively) from a single Fidl service is not necessarily
533  // restricted to a single .fidl file. Thus this map does not necessarily
534  // contain all of the types necessary to completely describe a service listed
535  // in the |services| map above. At runtime an implementation of
536  // |ServiceDescription| needs to merge the |RuntimeTypeInfo|
537  // from all of the generated modules in the file graph and then filter on only
538  // those types in the complete type set of a single service.
539  map<string, UserDefinedType> type_map;
540 };
