9 /*
10  * The structures in this file are intended to be used by the Fidl compiler
11  * and code generators. The front end of the compiler takes as input a
12  * .fidl file (or a list of .fidl files) and produces a FidlFileGraph struct.
13  *
14  * The backend of the compiler consumes a FidlFileGraph and invokes each of the
15  * code generators passing them data derived from the FidlFileGraph.
16  *
17  * A FidlFile represents the data parsed from a single .fidl file. Fidl
18  * files form a directed acyclic graph via the "imports" relation.
19  * That is, if file A imports file B then there is a directed edge in the
20  * graph from A to B. A FidlFileGraph represents the whole Graph.
21 
22  * The Fidl structures represented here have been fully resolved, meaning that
23  * the type references have been associated with their corresponding type
24  * definitions. This resolved type data is contained in the resolved_types data
25  * in FidlFileGraph.
26  */
27 
28 // A FidlFile represents the data defined by a single .fidl file, when
29 // all of the type references to types declared in imported .fidl files
30 // have been resolved.
31 struct FidlFile {
32  // |file_name| is (derived from) the file name of the corresponding
33  // .fidl file. It is the unique identifier for this module within the
34  // FidlFileGraph
35  string file_name;
36 
37  // |specified_file_name| is used to record information about the request that
38  // triggered the construction of the |FidlFileGraph| of which this
39  // |FidlFile| is a part. This field is populated by some producers (such
40  // as the Fidl compiler) but not necessarily all producers of this structure.
41  // If this field is null it means it is not supported by the producer.
42  //
43  // If this field is non-null but empty it means that this |FidlFile|
44  // was not explicitly requested but rather is included in the |FidlFileGraph|
45  // because it is referenced in the |imports| field of another |FidlFile|
46  // in the graph. If this field is non-empty it means that the corresponding
47  // .fidl file was explicitly requested, using |specified_file_name|. Note
48  // that in this case it is still possible that this file is also referenced
49  // in the |imports| field of another |FidlFile|.
50  string? specified_file_name;
51 
52  // The namespace is the identifier declared via the "module" declaration
53  // in the .fidl file.
54  string? module_namespace;
55 
56  // Attributes declared in the Fidl file at the module level.
57  array<Attribute>? attributes;
58 
59  // The list of other FidlFiles imported by this one. The elements
60  // of the array are the |file_name|s and the associated module may
61  // be retrieved from the FidlFileGraph.
62  array<string>? imports;
63 
64  // resolved_types and resolved_values in FidlFileGraph contain
65  // respectively the types and constants declared in the union of all modules
66  // in the graph. This KeysByType selects the keys of the types and constants
67  // defined in this module
68  KeysByType declared_fidl_objects;
69 
70  // The bytes encoding a |RuntimeTypeInfo| struct for this Fidl file,
71  // using Fidl message serialization. Some implementations may not include
72  // this. This string contains the base64 encoding of the gzip-compressed
73  // bytes.
74  string? serialized_runtime_type_info;
75 
76  // Comments attached to the file.
77  Comments? comments;
78 };
79 
80 // Represents a directed acyclic graph of FidlFiles.
81 struct FidlFileGraph {
82  // All the files in this graph. The keys are |file_name|s.
83  map<string, FidlFile> files;
84 
85  // All the resolved user-defined types in all the files in the graph. The keys are
86  // the |type_key|s.
87  map<string, UserDefinedType> resolved_types;
88 
89  // All the resolved DeclaredConstants in all the files in the graph. The keys are
90  // the |constant_key|s.
91  map<string, DeclaredConstant> resolved_constants;
92 };
93 
94 // A KeysByType struct organizes by type the keys of all types and consants in
95 // a FidlFile
96 struct KeysByType {
97  // The type keys of the types in the FidlFile.
98  array<string>? interfaces;
99  array<string>? structs;
100  array<string>? unions;
101  array<string>? top_level_enums;
102  array<string>? embedded_enums;
103 
104  // The constant keys of the constants in the FidlFile.
105  array<string>? top_level_constants;
106  array<string>? embedded_constants;
107 };
