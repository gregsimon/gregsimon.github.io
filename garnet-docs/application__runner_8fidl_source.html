41 // An interface for running applications.
42 //
43 // Typically exposed by applications that provide execution environments for
44 // particular classes of programs. For example, the Dart virtual machine exposes
45 // this interface to run Dart programs.
46 [ServiceName="app.ApplicationRunner"]
47 interface ApplicationRunner {
48  // Execute the given application.
49  //
50  // Upon startup, the application is to be given the information in
51  // |startup_info|, but the mechanism by which the application receives that
52  // information is up to the application runner.
53  //
54  // The |controller| interface request typically originates from the
55  // |ApplicationLauncher.CreateApplication| message that caused this
56  // application to be started.
57  StartApplication(ApplicationPackage application,
58  ApplicationStartupInfo startup_info,
59  ApplicationController&? controller);
60 };
