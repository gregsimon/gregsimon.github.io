11 // A Canvas allows clients to draw strokes, which are dynamically tessellated
12 // and inserted into the global Scenic scene-graph.
13 [ServiceName="sketchy::Canvas"]
14 interface Canvas {
15  // Init() must be called once, before invoking any other operations.
16  Init(CanvasListener? listener);
17 
18  // Enqueue a list of operations, which will not take effect until Present() is
19  // called. It is allowable to call Enqueue() several times before calling
20  // Present(); the effect is the same as if all operations had been
21  // concatenated into a single Enqueue() call.
22  //
23  // Akin to scenic::Session.Enqueue(); see scenic/session.fidl for details.
24  Enqueue(array<Op> ops);
25 
26  // Present all previously enqueued operations. In order to pipeline the
27  // preparation of the resources required to render the scene, two lists of
28  // fences (implemented as events) are passed.
29  //
30  // Akin to scenic::Session.Present(); see scenic/session.fidl for details.
31  Present(uint64 presentation_time) =>
32  (scenic.PresentationInfo presentation_info);
33 };
