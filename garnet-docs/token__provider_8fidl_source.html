25 // Stores attributes related to a firebase auth token for a given firebase api
26 // key. These tokens are obtained by agents using
27 // TokenProvider.GetFirebaseAuthToken().
28 //
29 // These tokens are minted by the Firebase Auth servers and are meant to be
30 // used for authorizing users into Firebase services such as DB and storage.
31 struct FirebaseToken {
32  // Use this as the auth token in firebase database and storage requests.
33  string id_token;
34 
35  // Use this to uniquely identify users.
36  string local_id;
37 
38  // Use this to uniquely identify user's email address provided by the
39  // Identity Provider.
40  string email;
41 };

43 // This interface provides oauth tokens for the currently logged-in user.
44 // This service can be obtained by Framework agents via
45 // AgentContext.GetTokenProvider().
46 interface TokenProvider {
47  // Gets the access token (with pre-defined scopes) for this user. Access token
48  // is returned from cache if not expired, otherwise refreshed from the server.
49  //
50  // Empty response is returned if running in incognito mode.
51  GetAccessToken() => (string? access_token, AuthErr auth_err);
52 
53  // Gets the Identity token (with pre-defined scopes) for this user. Identity
54  // token is returned from cache if not expired, otherwise refreshed from the
55  // server.
56  //
57  // Empty response is returned if running in incognito mode.
58  GetIdToken() => (string? id_token, AuthErr auth_err);
59 
60  // Gets the |FirebaseToken| for this user for the specified firebase api key.
61  // This api invokes firebase auth's VerifyAssertion endpoint that takes an
62  // OAuth IdToken as the input. Firebase auth token is returned from cache if
63  // not expired, otherwise refreshed from the server.
64  //
65  // Empty response is returned if running in incognito mode.
66  GetFirebaseAuthToken(string firebase_api_key) =>
67  (FirebaseToken? firebase_token, AuthErr auth_err);
68 
69  // Returns the client id used to fetch the above tokens.
70  GetClientId() => (string? client_id);
71 };
