52 // Instantiates a new resource.
53 struct CreateResourceOp {
54  // The ID of the resource to create; this ID must not currently be used within
55  // the session. The client may use this ID to refer to the resource in
56  // subsequent operations.
57  uint32 id;
58  // Specifies the type of resource to construct, as well as any other required
59  // data.
60  ResourceArgs args;
61 };
62 
63 // Releases the client's reference to the resource; it is then illegal to use
64 // the ID in subsequent Ops. Other references to the resource may exist, so
65 // releasing the resource does not result in its immediate destruction; it is
66 // only destroyed once the last reference is released. For example, after a
67 // stroke is added to a stroke-group, the group will keep the it alive even
68 // after the client releases the stroke.
69 //
70 // TODO: edge cases. What about releasing a stroke that is currently being
71 // drawn by Begin/Extend/FinishStrokeOp?
72 struct ReleaseResourceOp {
73  // The ID of the resource to be dereferenced. The ID must currently reference
74  // a resource (i.e. it must match one from a previous CreateResourceOp, and
75  // must not have match one from a subsequent ReleaseResourceOp).
76  uint32 id;
77 };

85 // Start drawing a stroke with a pen.
86 struct BeginStrokeOp {
87  // ID of the stroke to draw with the pen. This must be a new stroke, or one
88  // which has been cleared via ClearStrokeOp.
89  uint32 stroke_id;
90  // ID of the pen to draw the stroke with.
91  uint32 pen_id;
92  // Describes the time, position, etc. that the stroke was started.
93  Touch touch;
94 };
95 
96 // Continue drawing a stroke with a pen. The stroke must have previously been
97 // started (via BeginStrokeOp), but not finished.
98 struct ExtendStrokeOp {
99  // ID of the stroke that is being drawn.
100  uint32 stroke_id;
101  // ID of the pen that is drawing the stroke. Must match the ID provided
102  // to BeginStrokeOp.
103  uint32 pen_id;
104  // Touch events that describe the trajectory of the stroke since the previous
105  // BeginStrokeOp or ExtendStrokeOp.
106  array<Touch> touches;
107  // Procedurally-generated touch events that describe the predicted future
108  // trajectory of the stroke. Sketchy uses these to improve perceived latency
109  // while the stroke is being drawn, but they have no impact on the finished
110  // stroke.
111  // TODO: this may change to reflect however Mozart exposes predicted events.
112  array<Touch> predicted_touches;
113 };
114
