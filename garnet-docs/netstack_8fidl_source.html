64 [ServiceName="netstack::Netstack"]
65 interface Netstack {
66  // Finds the port number from a given service name and protocol. [service] can be a
67  // number like "42", or a service name like "http". If [protocol] is UNSPECIFIED,
68  // the service is checked for TCP first, then UDP.
69  GetPortForService(string service, Protocol protocol) => (uint16 port);
70 
71  // Finds the IP address for a given host name and port. This may issue network
72  // requests via DNS to look up domain names. E.g.
73  // GetAddress("example.com", 80) => [{142.42.42.1}]
74  GetAddress(string address, uint16 port) => (array<SocketAddress> addresses, NetErr err);
75 
76  // Returns the list of registered network interfaces.
77  GetInterfaces() => (array<NetInterface> interfaces);
78 
79  // Returns the netstack's node name.

81 
82  // TODO (porce): Separate interfaces.

84 };
