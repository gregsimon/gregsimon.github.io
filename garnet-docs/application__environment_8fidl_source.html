12 // An interface for managing a set of applications.
13 //
14 // Applications run inside environments, which provide ambient services and
15 // support for their lifecycle.
16 [ServiceName="app.ApplicationEnvironment"]
17 interface ApplicationEnvironment {
18  // Maximum length for an environment label.
19  const uint32 kLabelMaxLength = 32;
20 
21  // Creates a new environment nested inside this environment.
22  //
23  // When applications are created inside the nested environment using the
24  // environment's |ApplicationLauncher|, the environment requests the
25  // environment services from |host| before passing those services to
26  // the newly created application in its |ApplicationStartupInfo|.
27  //
28  // Typically the |host| will make the newly created |environment| available in
29  // the environment services, giving child applications to launch other
30  // applications in the environment and to create nested environments.
31  //
32  // The |host| must provide an |ApplicationLoader| service (or inherit one from
33  // its parent) if it wishes to allow new applications to be loaded in the
34  // nested environment.
35  //
36  // The |controller| can be used to control the lifecycle of the created
37  // environment. If an |ApplicationEnvironmentController|'s interface is
38  // request, the environment will be killed when the interface is closed.
39  //
40  // The |label| is an optional name to associate with the environment for
41  // diagnostic purposes. The label will be truncated if it is longer
42  // than |kLabelMaxLength|.
43  CreateNestedEnvironment(ApplicationEnvironmentHost host,
44  ApplicationEnvironment& environment,
45  ApplicationEnvironmentController&? controller,
46  string? label);
47 
48  // Gets the ApplicationLauncher associated with this environment.
49  //
50  // Applications created using this application launcher will be given the
51  // environment services provided by this environment's
52  // |ApplicationEnvironmentHost|.
53  GetApplicationLauncher(ApplicationLauncher& launcher);
54 
55  // Gets the services provided by this environment's
56  // |ApplicationEnvironmentHost|.
57  GetServices(ServiceProvider& services);
58 };
