16 // Represents the relationship between and subject timeline and a reference
17 // timeline.
18 //
19 // To translate a reference timeline value r to the subject timeline, apply
20 // the following formula:
21 //
22 // (r - reference_time) * subject_delta / reference_delta + subject_time
23 //
24 // To translate a subject timeline value s to the reference timeline, apply
25 // this formula provided subject_delta isn't zero:
26 //
27 // (s - subject_time) * reference_delta / subject_delta + reference_time
28 //
29 struct TimelineTransform {
30  // A value from the reference timeline that correlates to subject_time.
31  // When TimelineTransform is used in TimelineConsumer.SetTimelineTransform,
32  // reference_time is also the effective time at which the transform should
33  // be applied. In that case, it may be kUnspecifiedTime, to indicate that the
34  // transform should be applied as soon as possible.
35  int64 reference_time = 0;
36 
37  // A value from the subject timeline that correlates to reference_time. When
38  // TimelineTransform is used in TimelineConsumer.SetTimelineTransform,
39  // subject_time may be kUnspecifiedTime to indicate that the transition
40  // should be first-order continuous. That is, kUnspecifiedTime indicates that
41  // the consumer should use the previous timeline transform to calculate the
42  // correct subject_time from the reference_time (if specified) or the
43  // reference time chosen by the timeline consumer if reference_time is not
44  // specified.
45  int64 subject_time = 0;
46 
47  // The change in the reference timeline corresponding to subject_delta.
48  // Cannot be zero.
49  uint32 reference_delta = 1;
50 
51  // The change in the subject timeline corresponding to reference_delta.
52  uint32 subject_delta = 0;
53 };
54 
55 // A push-mode consumer of timeline updates.
56 interface TimelineConsumer {
57  // Sets a timeline transform. Note that in this usage, the reference_time and
58  // subject_time fields of TimelineTransform have semantics beyond being a
59  // correlated pair of time values, and either or both may be unspecified
60  // (kUnspecifiedTime).
61  //
62  // The reference_time is the effective time (in the reference timeline) at
63  // which the new transform is to be applied. If not specified, the transition
64  // is to occur as soon as possible, and the consumer must choose the earliest
65  // reference time at which the transition can reliably occur.
66  //
67  // The subject_time is the new subject time that will correlate to the
68  // specified or consumer-chosen reference time. If not specified, the
69  // consumer uses the previous timeline transform to calculate a subject time
70  // that will achieve a first-order continuous transition. If specified, the
71  // timeline will effectively jump to that value at the chosen reference time,
72  // a transition that's unlikely to be first-order continuous.
73  //
74  // The callback is called at the effective time or when a pending operation is
75  // cancelled due to a subsequent call, in which case the 'completed' value is
76  // false.
77  SetTimelineTransform(TimelineTransform timeline_transform) =>
78  (bool completed);
79 };
